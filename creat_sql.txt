-- Storage bucket: cv-images
-- Run once in the Supabase SQL editor (the storage schema must exist).
-- ===========================================================

insert into storage.buckets (id, name, public)
values ('cv-images', 'cv-images', true)
on conflict (id) do nothing;

-- Public can read images
create policy "Public can read cv-images"
  on storage.objects for select
  using (bucket_id = 'cv-images');

-- Authenticated users can upload to cv-images
create policy "Authenticated can upload cv-images"
  on storage.objects for insert
  to authenticated
  with check (bucket_id = 'cv-images');

-- Owner can update / delete their uploads
create policy "Owner can update cv-images"
  on storage.objects for update
  to authenticated
  using (bucket_id = 'cv-images' and auth.jwt() ->> 'email' = 'yuqi.guo17@gmail.com');

create policy "Owner can delete cv-images"
  on storage.objects for delete
  to authenticated
  using (bucket_id = 'cv-images' and auth.jwt() ->> 'email' = 'yuqi.guo17@gmail.com');

-- ============================================================
-- visitor_logs: 启用 RLS，禁止 anon 直接 insert
-- ============================================================
alter table public.visitor_logs enable row level security;

-- anon 可以 select（DashboardPanels 读取需要）
create policy "Anyone can read visitor_logs"
  on public.visitor_logs for select
  using (true);

-- Explicitly deny direct inserts from the anon role.
-- All inserts MUST go through the API route which uses the service_role key
-- (service_role bypasses RLS entirely, so it is unaffected by this policy).
--
-- IMPORTANT: must be AS RESTRICTIVE.
-- A plain (permissive) WITH CHECK (false) is OR-ed with every other
-- permissive INSERT policy, so any existing allow-policy would win.
-- AS RESTRICTIVE is AND-ed with all permissive results — it always wins.
drop policy if exists "Deny anon inserts on visitor_logs" on public.visitor_logs;
create policy "Deny anon inserts on visitor_logs"
  on public.visitor_logs
  as restrictive
  for insert
  to anon
  with check (false);

-- Drop any pre-existing permissive insert policies
drop policy if exists "Allow public inserts on visitor_logs" on public.visitor_logs;
